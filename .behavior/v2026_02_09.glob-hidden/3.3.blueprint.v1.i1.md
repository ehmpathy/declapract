# blueprint: glob matches files in nested hidden directories

## summary

switch from `fast-glob` to node.js built-in `fs.glob` (node 22+) to fix globstar traversal of hidden directories. the fix is native and requires no workarounds.

## implementation details

### evaluateProjectAgainstFileCheckDeclaration.ts

**before**:
```ts
import glob from 'fast-glob';

const pathsFoundByGlob = await withDurationReport(
  `glob:${check.pathGlob}`,
  () => glob(check.pathGlob, {
    cwd: project.getProjectRootDirectory(),
    dot: true,
    onlyFiles: true,
    ignore: ['node_modules'],
  }),
)();
```

**after**:
```ts
import { glob } from 'node:fs/promises';
import { stat } from 'node:fs/promises';
import { join } from 'node:path';

const pathsFoundByGlob = await withDurationReport(
  `glob:${check.pathGlob}`,
  async () => {
    const cwd = project.getProjectRootDirectory();
    const matches: string[] = [];
    for await (const entry of glob(check.pathGlob, {
      cwd,
      exclude: (name) => name === 'node_modules',
    })) {
      // filter to files only (fs.glob lacks onlyFiles option)
      const fullPath = join(cwd, entry);
      const stats = await stat(fullPath);
      if (stats.isFile()) matches.push(entry);
    }
    return matches;
  },
)();
```

### listFilesInDirectory.ts

**before**:
```ts
import glob from 'fast-glob';

export const listFilesInDirectory = async ({
  directory,
}: {
  directory: string;
}) => glob('**/*', { cwd: directory, dot: true, onlyFiles: true });
```

**after**:
```ts
import { glob, stat } from 'node:fs/promises';
import { join } from 'node:path';

export const listFilesInDirectory = async ({
  directory,
}: {
  directory: string;
}): Promise<string[]> => {
  const matches: string[] = [];
  for await (const entry of glob('**/*', { cwd: directory })) {
    const fullPath = join(directory, entry);
    const stats = await stat(fullPath);
    if (stats.isFile()) matches.push(entry);
  }
  return matches;
};
```
